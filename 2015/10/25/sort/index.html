选择排序和冒泡排序在python下的实现

  1. 选择排序：

    import sys
def sort(a):
    alen=len(a)
    for i in range(alen):
        min=i
        for j in range(i+1,alen):
            if a[j]<a[min]:
                min=j
            if min!=i:
                a[i],a[min]=a[min],a[i]
if __name__=='__main__':
    x=[12,34,78,3,7,5]
    print 'before sort',x
    sort(x)
    print 'after sort', x

  2. 冒泡排序：

    import sys
def sort(a):
    alen=len(a)
    for i in range(alen):
        for j in range(alen-1):
            if a[j]>a[j+1]:
                a[j],a[j+1]=a[j+1],a[j]

if __name__=='__main__':
    b=[33,12,23,67,2,0,1]
    print 'before sort',x
    sort(b)
    print 'after sort :',b
